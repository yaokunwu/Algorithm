## Graph相关，adjacency list
4.  一个没有重复数字的数组，将这个数组所有相邻的数，打乱顺序生成一个int[2]放进一个array里面, 再打乱顺序给你，让你返回原数组。(倒序也是正确的) <br>
例子: [3, 5, 1, 2, 4]<br>
有打乱顺序的相邻pair [[4,2],[1,2], [5,3],[5,1]] 返回原数组<br>
思路: 建立adjacency list然后dfs或bfs依次连接
```Java
Class Solution {
    Map<Integer, List<Integer>> adjList;
    public List<Integer> canFinish(int[][] pairs) {
      Map<Integer, Integer> inDegree = new HashMap<>();
      adjList = new HashMap<>();
      for (int[] pair : pairs) {
          int val1 = pair[0];
          int val2 = pair[1];
          inDegree.put(val1, inDegree.getOrDefault(val1, 0) + 1);
          inDegree.put(val2, inDegree.getOrDefault(val2, 0) + 1);
          if (adjList.containsKey(val1)) {
              List<Integer> curr = adjList.get(val1);
              curr.add(val2);
          } else {
              adjList.put(val1, new ArrayList<>());
              adjList.get(val1).add(val2);
          }
          if (adjList.containsKey(val2)) {
              List<Integer> curr = adjList.get(val2);
              curr.add(val1);
          } else {
              adjList.put(val2, new ArrayList<>());
              adjList.get(val2).add(val1);
          }
      }
      List<Integer> res = new ArrayList<>();
      int currVal = 0;
      for (Map.Entry<Integer, Integer> entry : inDegree.entrySet()) {
          if (entry.getValue() == 1) {
              currVal = entry.getKey();
              break;
          }
      }
      Integer prev = null;
      dfs(res, currVal, prev);
      return res;
    }

    private void dfs(List<Integer> res, int val, Integer prev) {
      res.add(val);
      List<Integer> childs = adjList.get(val);
      for (int child : childs) {
          if (prev == null) {
              dfs(res, child, val);
          }
          if (prev != null && child != prev) {
              dfs(res, child, val);
          }
      }
    }
}
```

## Merge Interval
Given a list of intervals, merge all the overlapping intervals to produce a list that has only mutually exclusive intervals.

**Algorithm**
1. Traverse the list of intervals;
2. Create a stack to main the previous interval
3. if overlapping happens, combine the interval and going to the next interval
4. pop the last item and add to result
```Java
Class Solution {
    private Class Interval {
        int start, int end;
        Interval(int start, int end) {
            this.start = start;
            this.end = end;
        }
    }
    public List<Interval> merge(List<Integer> intervals) {
        List<Integer> res = new ArrayList<>();
        Deque<Integer> stack = new LinkedList<>();
        for (Interval interval : intervals) {
            if (!stack.isEmpty()) {
                stack.push(interval);
            } else {
                if (interval.start <= stack.peek().start) {
                    Interval prev = stack.pop();
                    Interval merged = new Interval(prev.start,
                    Math.max(prev.end, interval.end));
                    stack.push(merged);
                } else {
                    res.add(stack.pop());
                    stack.push(interval);
                }
            }
        }
        if (!stack.isEmpty()) {
            res.add(stack.pop());
        }
        return res;
    }
}
```

### Insert Interval
**Algorithm**
1. traverse the list of Interval
2. found the first interval that needs to be merged, make the merge and process
   the rest;
```Java
    public List<Interval> insert(List<Interval> intervals, Interval interval) {
        List<Interval> res = new ArrayList<>();
        if (intervals == null || intervals.length() < 1) {
            res.add(interval);
            return res;
        }
        if (interval.end < intervals.get(0).start) {
            res.add(interval);
            res.addAll(intervals);
            return res;
        }
        if (interval.start > intervals.get(interval.length() - 1).end) {
            res.addAll(intervals);
            res.add(interval);
            return res;
        }
        Deque<Interval> stack = new LinkedList<>();
        int idx = 0;
        for (; idx < intervals.length; idx++) {
            Interval curr = intervals.get(idx);
            if (interval.start > curr.end) {
                res.add(curr);
            } else {
                break;
            }
        }
        Interval merged = new Interval(Math.min(intervals.get(idx).start,
        interval.start), Math.max(intervals.get(idx).end, interval.end));
        stack.push(merged);
        idx++;
        for (;idx < intervals.length(); idx++) {
            Interval curr = intervals.get(idx);
            if (curr.start <= stack.peek().end) {
                Interval prev = stack.pop();
                Interval merged = new Interval(prev.start, Math.max(prev.end,
                curr.end));
                stack.push(merged);
            } else {
                res.add(stack.pop());
                stack.push(curr);
            }
        }
        if (!stack.isEmpty()) {
            res.add(stack.pop());
        }
        return res;
    }
```

### Interval Intersection
**Algorithm**
1. Start from the first interval in the two intersections
2. find the intersection and added to res and go to the next item with smaller end until one of them reach to the end.
```Java
public static Interval[] merge(Interval[] arr1, Interval[] arr2) {
    if (arr1 == null || arr1.length == 0 || arr2 == null || arr2.length == 0) {
        return null;
    }
    List<Interval> res = new ArrayList<>();
    int i = 0, j = 0;
    while (i < arr1.length && j < arr2.length) {
        Interval inter1 = arr1[i];
        Interval inter2 = arr2[j];
        int start = Math.max(inter1.start, inter2.start);
        int end = Math.min(inter1.end, inter2.end);
        if (start <= end) {
            res.add(new Interval(start, end));
        }
        if (inter1.end < inter2.end) {
            i++;
        } else {
            j++;
        }
    }
    return res.toArray(new Interval[res.size()]);
}
```

```Java
public static int findMinimumMeetingRooms(List<Meeting> meetings) {
    if (meetings == null || meetings.size() == 0) {
        return 0;
    }
    Collections.sort(meetings, (a, b) -> a.end - b.end);
    PriorityQueue<Meeting> minHeap = new PriorityQueue<>((a, b) -> a.end - b.end);
    minHeap.offer(meetings.get(0));
    int minSize = 1;
    for (int i = 1; i < meetings.size(); i++) {
        while (meetings.get(i).start >= minHeap.peek().end) {
            minHeap.pop();
        }
        minHeap.offer(meetings.get(i));
        minSize = Math.max(minSize, minHeap.size());
    }
    return minSize;
}
```

### Max CPU Load
**Algorithm**
1. Init a priorityqueue to store the processes that are running simutianously
2. At any position, remove the ending task and insert the new task and calculate
   the CPU load at that moment and updata the Maximum value
3. return the maximum

```Java
public static int findMaxCPULoad(List<Job> jobs) {
    int maxLoad = 0;
    int currLoad = 0;
    if (jobs == null || jobs.isEmpty()) {
        return maxLoad;
    }
    Collections.sort(jobs, (a, b) -> a.start - b.start);
    PriorityQueue<Job> minHeap = new PriorityQueue<>((a, b) -> a.end - b.end);
    for (Job job : jobs) {
        while(!minHeap.isEmpty() && job.start >= minHeap.peek().end) {
            Job endingTask = minHeap.poll();
            currLoad -= endingTask.cpuLoad;
        }
        minHeap.offer(job);
        currLoad += job.cpuLoad;
        maxLoad = Math.max(maxLoad, currLoad);
    }
    return maxLoad;
}
```

```
public static List<Interval> findEmployeeFreeTime(List<List<Interval>> schedule) {
       List<Interval> res = new ArrayList<>();
       PriorityQueue<Node> minHeap = new PriorityQueue<>((a, b) ->
       schedule.get(a.id).get(a.idx).start - schedule.get(b.id).get(b.idx).start);
       for (int i = 0; i < schedule.size(); i++) {
           minHeap.offer(new Node(i, 0));
       }
       
       while(minHeap.size() > 1) {
          Node currNode = minHeap.poll();
          Node top = minHeap.peek();
          int start = schedule.get(top.id).get(top.idx).start;
          int end = schedule.get(currNode.id).get(currNode.idx).end;
          if (start > end) {
              res.add(new Interval(end, start));
          }
          int nextIdx = currNode.idx + 1;
          if (nextIdx < schedule.get(currNode.id).size()) {
              minHeap.offer(new Node(currNode.id, nextIdx));
          }
      }
      Node left = minHeap.poll();
      List<Interval> rest = schedule.get(left.id);
      Interval prev = rest.get(left.idx);
      for (int i = left.idx + 1; i < rest.size(); i++) {
           Interval curr = new Interval(prev.end, rest.get(i).start);
            res.add(curr);
           prev = curr;
      }
      
      return res;
}
```

### Merging problem
**Algorithm**
1. Create a minHeap that takes the first item from each list and tracking the
   largest value
2. at each iteration, check the current length of the range and update the res.
   remove the smallest item and add the next item
3. if any list reach to the end, stop the loop;
```Java

class Node {
    int eleId id;
    Node(int id, int eleId) {
        this.eleId = eleId;
        this.id = id;
    }
};
public static int[] findSmallestRange(List<Integer[]> lists) {
    int[] res = new int[2];
    int len = Integer.MAX_VALUE;
    int maxVal = Integer.MIN_VALUE;
    if (lists == null || lists.size() == 0) {
        return new int[0];
    }
    PriorityQueue<Node> minHeap = new PriorityQueue<>((a, b) -> a - b);
    for (int i = 0; i < lists.size(); i++) {
        int value = lists.get(i)[0];
        maxVal = Math.max(maxVal, value);
        minHeap.offer(new Node(i, 0));
    }
    while (minHeap.size() == lists.size()) {
        Node currNode = minHeap.poll();
        if (maxVal - lists.get(currNode.id)[currNode.eleId] < len) {
            len = maxVal - lists.get(currNode.id)[currNode.eleId];
            res[0] = lists.get(currNode.id)[currNode.eleId];
            res[1] = maxVal;
        }
        int currId = currNode.eleId;
        if (currId + 1 < lists.get(currNode.id)) {
            minHeap.offer(new Node(currNode.id, currId + 1));
            maxVal = Math.max(maxval, lists.get(currNode.id)[currId + 1])
        }
    }
    return res;
}
```

### K Pairs with Largest Sums
**Algorithm**
1. iterate throught the two lists before k or the end
2. add the pairs to the res until we have k pairs in the result
3. update the pairs if we found a bigger sum, skip the rest if the two sum less
   than the top element sum
```Java
public static List<int[]> findKLargestPairs(int[] nums1, int[] nums2, int k) {
    List<int[]> res = new ArrayList<>();
    PriorityQueue<int[]> minHeap = new PriorityQueue<>((a, b) ->(a[0] + a[1]) -
    (b[0] + b[1]));
    for (int i = 0; i < nums1.length && i < k; i++) {
       for (int j = 0; j < num2.length && j < k; j++) {
            if (minHeap.size() < k){
                minHeap.offer(new int[] {nums1[i], nums2[j]});
            } else {
                if (nums1[i] + nums2[j] < minHeap.peek()[0] + minHeap.peek()[1])
                {
                    break;
                
                }
                minHeap.poll();
                minHeap.offer(new int[] {nums1[i], nums2[j]);
            }
       }
       
    } 
    while (!minHeap.isEmpty()) {
        res.add(minHeap.poll());
    }
    return res;
}
```

### Topological sort
**Algorithm**
1. this is a directed graph where the input are given by the directional pairs
2. traverse the edge list and calculate the indegree for each vertex and create
   the adjacency list
3. starting at the vertex which indegree is zero
4. using a queue to monitor the sequence, process the vertex and find the new
   vertex which indegree is zero, repeat the process
```Java
public static List<Integer> sort(int vertices, int[][] edges) {
    Map<Integer, Integer> indegree = new HashMap<>();
    Map<Integer, List<Integer>> adjList = new HashMap<>();
    for (int[] edge : edges) {
       int parent = edge[0];
       int child = edge[1];
       if (adjList.containsKey(parent)) {
          List<Integer> curr = adjList.get(parent);
          curr.add(child);
       } else {
          List<Integer> curr = new ArrayList<>();
              curr.add(child);
              adjList.put(parent, curr);
         }
          indegree.put(child, indegree.getOrDefault(child, 0) + 1);
          if (!indegree.containsKey(parent)) {
            indegree.put(parent, 0);
          }
      }
      Queue<Integer> queue = new LinkedList<>();
      for (Map.Entry<Integer, Integer> entry : indegree.entrySet()) {
          if (entry.getValue() == 0) {
              queue.offer(entry.getKey());
          }
      }
      List<Integer> res = new ArrayList<>();
      while (!queue.isEmpty()) {
         int curr = queue.poll();
          res.add(curr);
          if (adjList.get(curr) != null) {
            for (int child : adjList.get(curr)) {
              indegree.put(child, indegree.get(child) - 1);
               if (indegree.get(child) == 0) {
                   queue.offer(child);
               }
           }
          }
          
    }
     if (res.size() != vertices) {
          return null;
      } 
      return res;
}
``` 
 
 
### Find all possible schedualing
**Algorithm**
Backtracking and topological sort
1. Similar to the previous problem, we first construct the indegree map and
   adjacency list
2. at each step, use a dfs to get possible combination and backtracking
3. at the end of dfs, print the possible solution
```Java
int[] visited;

int tasks;
Map<Integer, List<Integer>> adjList;
public static void printOrders(int tasks, int[][] prerequisites) {
    visited = new int[tasks];
    indegree = new HashMap<>();
    this.tasks = tasks;
    adjList = new HashMap<>();
    for (int i = 0; i < tasks; i++) {
        indegree.put(i, 0);
        adjList.put(i, new ArrayList<>());
    }
    for (int[] edge : prerequisites) {
        int parent = edge[0];
        int child = edge[1];
        indegree.put(child, indegree.get(child) + 1);
        List<Integer> curr = adjList.get(parent);
        curr.add(child);
    }
    dfs(0, new ArrayList<>());    
}

private void dfs(int level, List<Integer> curr) {
    if (level == tasks) {
        System.out.println(curr);
        return;
    }
    List<Integer> choices = new ArrayList<>();
    for (Map.Entry<Integer, Integer> entry : indegree.entrySet()) {
        if (entry.getValue() == 0 && visited[entry.getKey()] == false) {
            choices.add(entry.getKey());
        }
    }
    for (Integer key : choices) {
        curr.add(key);
        for (Integer child : adjList.get(key)) {
            indegree.put(child, indegree.get(child) - 1);
         
        }
        visited[key] = true;
        dfs(level + 1, curr);
        for (Integer child : adjList.get(key)) {
            indegree.put(child, indegree.get(child) + 1);
        }
        visited[key] = false;
        curr.remove(curr.size() - 1);
    }
}
```

```Java
class Solution {
    int maxLen = Integer.MIN_VALUE;
    TreeSet<String> res;
    Set<String> visited;
    Map<String, Set<String>> adjList;
    public String[] findMaxGroup(String[][] itemList) {
        adjList = new HashMap<>();
        visited = new HashSet<>();

        for (String[] item : itemList) {
            String left = item[0];
            String right = item[1];
            if (!adjList.containsKey(left)) {
                adjList.put(left, new HashSet<String>());

            }
            if (!adjList.containsKey(right)) {
                adjList.put(right,new HashSet<String>());
            }
        }

        for (String[] item : itemList) {
            String left = item[0];
            String right = item[1];
            adjList.get(left).add(right);
            adjList.get(right).add(left);
        }
        for (Map.Entry<String, Set<String>> entry : adjList.entrySet()) {
            if (!visited.contains(entry.getKey())) {
                TreeSet<String> currList = new TreeSet<>();
                visited.add(entry.getKey());
                bfs(entry.getKey(), currList);
            }
        }

        return res.toArray(new String[res.size()]);
    }

    private void bfs(String currKey, TreeSet<String> currList) {
        Queue<String> queue = new LinkedList<>();
        queue.offer(currKey);
        int count = 0;
        while (!queue.isEmpty()) {
            String curr = queue.poll();
            currList.add(curr);
            count++;
            for (String item : adjList.get(curr)) {
                if (!visited.contains(item)) {
                    queue.offer(item);
                    visited.add(item);
                }
            }
        }
        if (count > maxLen) {
            maxLen = count;
            res = new TreeSet<>(currList);
        } else if (count == maxLen && currList.first().compareTo(res.first()) > 0) {
            res = new TreeSet<>(currList);
        }
    }
}
```

```Java
class Solution {
    public String convertToTitle(int n) {
        StringBuilder sb = new StringBuilder();
        while (n != 0) {
            n--;
            int curr = n % 26;
            char currChar = (char) ('A' + curr);
            sb.append(currChar);
            n = n / 26;
        }
        return sb.reverse().toString();
    }
}
```

```Java
        //Algorithm

        int res = 0;
```
